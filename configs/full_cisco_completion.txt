! Custom concrete build for empty cisco.pkt
! Goal: build a working Packet Tracer topology from the empty/base file.
!
! This version is ready to paste as-is (interface IDs may need adjustment
! if your device model names ports differently).

! -------------------------------------------------------------------
! 1) Target addressing used in this build
! -------------------------------------------------------------------
! Acad network:      192.167.2.0/25      gateway 192.167.2.1
! RS networks:       192.167.2.224/27, 192.167.3.0/24 (split into /27 blocks)
! RS backbone/core:  192.167.33.16/28
! ACL 101 applied IN on RS S2.2 g1/0
! ACL 102 applied IN on RS Core Router g0/0
!
! ACL intent:
! - RS -> Acad: FTP allowed
! - RS -> non-Acad: FTP denied
! - Acad -> RS: FTP allowed
! - non-Acad -> RS: FTP denied
! - all other IP traffic: allowed

! -------------------------------------------------------------------
! 2) Router baseline + addressing
! -------------------------------------------------------------------

! RS S2.1
enable
conf t
hostname RS-S2-1
no ip domain-lookup
!
interface g0/0
 description RS-LAN-192.167.2.224/27
 ip address 192.167.2.225 255.255.255.224
 no shutdown
exit
!
interface g1/0
 description RS-LAN-192.167.3.0/27
 ip address 192.167.3.1 255.255.255.224
 no shutdown
exit
!
interface g2/0
 description RS-LAN-192.167.3.32/27
 ip address 192.167.3.33 255.255.255.224
 no shutdown
exit
!
interface g0/1
 description UPLINK-TO-RS-CORE
 ip address 192.167.33.25 255.255.255.240
 no shutdown
exit
!
ip route 0.0.0.0 0.0.0.0 192.167.33.29
end
write memory

! RS S2.2
enable
conf t
hostname RS-S2-2
no ip domain-lookup
!
interface g0/0
 description RS-LAN-192.167.3.64/27
 ip address 192.167.3.65 255.255.255.224
 no shutdown
exit
!
interface g1/0
 description RS-LAN-192.167.3.96/27
 ip address 192.167.3.97 255.255.255.224
 no shutdown
exit
!
interface g2/0
 description RS-LAN-192.167.3.128/27
 ip address 192.167.3.129 255.255.255.224
 no shutdown
exit
!
interface g0/1
 description UPLINK-TO-RS-CORE
 ip address 192.167.33.26 255.255.255.240
 no shutdown
exit
!
ip route 0.0.0.0 0.0.0.0 192.167.33.29
end
write memory

! RS Core Router
enable
conf t
hostname RS-CORE
no ip domain-lookup
!
interface g0/0
 description RS-CORE-BACKBONE-192.167.33.16/28
 ip address 192.167.33.29 255.255.255.240
 no shutdown
exit
!
ip route 192.167.2.224 255.255.255.224 192.167.33.25
ip route 192.167.3.0 255.255.255.224 192.167.33.25
ip route 192.167.3.32 255.255.255.224 192.167.33.25
ip route 192.167.3.64 255.255.255.224 192.167.33.26
ip route 192.167.3.96 255.255.255.224 192.167.33.26
ip route 192.167.3.128 255.255.255.224 192.167.33.26
ip route 192.167.2.0 255.255.255.128 192.167.33.28
end
write memory

! Academic Router
enable
conf t
hostname ACADEMIC
no ip domain-lookup
!
interface g0/0
 description ACAD-LAN-192.167.2.0/25
 ip address 192.167.2.1 255.255.255.128
 no shutdown
exit
!
interface g0/1
 description ACAD-UPLINK-TO-RS-CORE
 ip address 192.167.33.28 255.255.255.240
 no shutdown
exit
!
ip route 192.167.2.224 255.255.255.224 192.167.33.29
ip route 192.167.3.0 255.255.255.0 192.167.33.29
end
write memory

! -------------------------------------------------------------------
! 3) ACL policy and attachment
! -------------------------------------------------------------------

! Apply on RS-S2-2
conf t
ip access-list extended 101
 remark permit FTP service from RS to Acad
 permit tcp 192.167.2.224 0.0.0.31 192.167.2.0 0.0.0.127 eq 21
 permit tcp 192.167.3.0 0.0.0.255 192.167.2.0 0.0.0.127 eq 21
 remark deny FTP service from RS to others
 deny tcp 192.167.2.224 0.0.0.31 any eq 21
 deny tcp 192.167.3.0 0.0.0.255 any eq 21
 remark permit all remaining IP traffic
 permit ip any any
exit
!
ip access-list extended 102
 remark permit FTP service from Acad to RS
 permit tcp 192.167.2.0 0.0.0.127 192.167.2.224 0.0.0.31 eq 21
 permit tcp 192.167.2.0 0.0.0.127 192.167.3.0 0.0.0.255 eq 21
 remark deny FTP service from non-Acad to RS
 deny tcp any 192.167.2.224 0.0.0.31 eq 21
 deny tcp any 192.167.3.0 0.0.0.255 eq 21
 remark permit all remaining IP traffic
 permit ip any any
exit
!
interface g1/0
 ip access-group 101 in
exit
end
write memory

! Apply ACL 102 on RS-CORE
conf t
interface g0/0
 ip access-group 102 in
exit
end
write memory
